```typescript
  /**
   * 零 JS 读取 document 滚动状态（收口出口）。
   * 用 CDP Page.getLayoutMetrics：layoutViewport.pageY=scrollY，clientHeight=视口高，
   * contentSize.height=文档总高。供 crawler 层到顶/到底判定，禁止裸用 page.evaluate。
   * 失败返回 null（遵循铁律，不静默降级）。
   */
  static async cdpGetDocumentScrollState(page: Page): Promise<{ scrollY: number; clientHeight: number; scrollHeight: number } | null> {
    try {
      return await HumanActions.withCDPContext(page, async (ctx) => {
        const m = await ctx.cdp.getLayoutMetrics();
        return {
          scrollY: m.layoutViewport.pageY,
          clientHeight: m.layoutViewport.clientHeight,
          scrollHeight: m.contentSize.height,
        };
      });
    } catch (err: any) {
      logger.warn({ err: err.message }, '[cdpGetDocumentScrollState] read failed');
      return null;
    }
  }
```
